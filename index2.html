Startup Brightness

>> sudo -H gedit /etc/rc.local

>> echo X > /sys/class/backlight/intel_backlight/brightness

>> sudo gedit /usr/bin/dropbox

>> os.environ['DBUS_SESSION_BUS_ADDRESS'] = ""
Save and reboot
